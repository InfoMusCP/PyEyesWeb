Impulsivity

Impulsivity evaluation based on direction change and suddenness.

This module implements the Impulsivity mid-level feature. Impulsivity measures the tendency of a movement to be both abrupt in direction and sudden in execution.

It combines the DirectionChange and Suddenness features to provide a single metric reflecting impulsive motor behavior.

class Impulsivity(DynamicFeature):
    """Compute the impulsivity of movement within a time window.

    Impulsivity is defined as the product of the **direction change** (geometric
    curvature) and the **suddenness** (stable distribution index).

    Read more in the [User Guide](../../user_guide/theoretical_framework/mid_level/mid_level.md).

    Parameters
    ----------
    suddenness_algo : {"new", "old"}, optional
        Algorithm variant used for suddenness detection.  Defaults to `"new"`.
    """

    def __init__(self, suddenness_algo: str = "new"):
        super().__init__()
        self._dc_analyzer = DirectionChange(method="area")
        self._suddenness_analyzer = Suddenness(algo=suddenness_algo)

    def compute(self, window_data: np.ndarray, **kwargs) -> ImpulsivityResult:
        r"""Compute impulsivity score for the given window.

        Parameters
        ----------
        window_data : numpy.ndarray of shape (Time, N_signals, N_dims)
            Window of motion data.
        **kwargs
            Unused; accepted for API compatibility.

        Returns
        -------
        ImpulsivityResult
            Result containing the impulsivity score.  Returns
            `ImpulsivityResult(is_valid=False)` when fewer than 5 samples
            are available.

        Notes
        -----
        The impulsivity score is computed as:

        $$
        \text{impulsivity} = \text{poly_area} \times (1 - \alpha)
        $$

        where `poly_area` comes from [DirectionChange][pyeyesweb.low_level.direction_change.DirectionChange]
        and $\alpha$ is the stability index from [Suddenness][pyeyesweb.mid_level.suddenness.Suddenness].
        """
        n_samples = window_data.shape[0]
        if n_samples < 5:
            return ImpulsivityResult(is_valid=False)

        # 1. Direction Change (Area)
        dc_res = self._dc_analyzer.compute(window_data)
        if not dc_res.is_valid or dc_res.polygon_area is None:
            return ImpulsivityResult(is_valid=False)

        # 2. Suddenness
        sud_res = self._suddenness_analyzer.compute(window_data)
        if not sud_res.is_valid:
            return ImpulsivityResult(is_valid=False)

        # 3. Combine them
        # Lower alpha means more impulsive (heavier tails)
        # poly_area measures curvature
        impulsivity = dc_res.polygon_area * (1.0 - sud_res.alpha)

        return ImpulsivityResult(impulsivity=float(max(0.0, impulsivity)))
