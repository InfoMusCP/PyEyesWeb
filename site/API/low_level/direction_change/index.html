Direction Change

Direction change analysis for temporal movement trajectories.

This module provides metrics for detecting direction changes in movement. It evaluates how much a trajectory departs from a straight line using two complementary methods:

  • Cosine Similarity — measures the angle between successive velocity vectors.
  • Polygon Area — measures the area of the polygon formed by three consecutive positions (a measure of curvature).

class DirectionChange(DynamicFeature):
    """Detect changes in movement direction using angular and geometric metrics.

    Analyses the trajectory of the **first signal** (index 0) in the input window.
    It computes the consistency of its velocity direction over time.

    Read more in the [User Guide](../../user_guide/theoretical_framework/low_level/low_level.md).

    Parameters
    ----------
    method : {"cosine", "area", "both"}, optional
        Algorithm variant to use.  Defaults to `"both"`.

    Examples
    --------
    >>> dc = DirectionChange(method="cosine")
    """

    def __init__(self, method: str = "both"):
        super().__init__()
        self.method = method

    def _cosine_similarity(self, velocities: np.ndarray) -> Optional[float]:
        if velocities.shape[0] < 2:
            return None

        v1 = velocities[:-1]
        v2 = velocities[1:]

        # Element-wise dot products across dimensions (axis 1)
        dot = np.sum(v1 * v2, axis=1)

        # Norms
        norm1 = np.linalg.norm(v1, axis=1)
        norm2 = np.linalg.norm(v2, axis=1)

        # Cosine similarity excluding zero-velocity cases
        mask = (norm1 > 1e-10) & (norm2 > 1e-10)
        if not np.any(mask):
            return 1.0

        cos_sim = dot[mask] / (norm1[mask] * norm2[mask])
        return float(np.mean(cos_sim))

    def _polygon_area(self, positions: np.ndarray) -> Optional[float]:
        if positions.shape[0] < 3:
            return None

        # Triplet of points: P1, P2, P3
        p1 = positions[:-2]
        p2 = positions[1:-1]
        p3 = positions[2:]

        # Vectors P1->P2 and P1->P3
        v1 = p2 - p1
        v2 = p3 - p1

        # Triangle area = 0.5 * ||v1 x v2||
        # cross() handles both 2D (scalar) and 3D (vector) cases correctly
        cross_prod = np.cross(v1, v2)

        if cross_prod.ndim == 1:  # 2D case result is scalar
            areas = 0.5 * np.abs(cross_prod)
        else:  # 3D case result is vector
            areas = 0.5 * np.linalg.norm(cross_prod, axis=1)

        return float(np.sum(areas) / len(areas))

    def compute(self, window_data: np.ndarray) -> DirectionChangeResult:
        """Compute angular and geometric direction change metrics.

        Parameters
        ----------
        window_data : numpy.ndarray of shape (Time, N_signals, N_dims)
            Window of motion data.  The first signal's trajectory is used.

        Returns
        -------
        DirectionChangeResult
            Result containing the requested direction change metrics.
            Returns `DirectionChangeResult(is_valid=False)` if the window
            contains fewer than 2 (for cosine) or 3 (for area) points.
        """
        # Trajectory of the first joint
        pos = window_data[:, 0, :]

        if pos.shape[0] < 2:
            return DirectionChangeResult(is_valid=False)

        cos_sim = None
        poly_area = None

        if self.method in ["cosine", "both"]:
            # Approximate velocities via finite differences
            vel = pos[1:] - pos[:-1]
            cos_sim = self._cosine_similarity(vel)

        if self.method in ["area", "both"]:
            poly_area = self._polygon_area(pos)

        # If we asked for a method that requires more points than we have, it might be None
        if cos_sim is None and poly_area is None:
            return DirectionChangeResult(is_valid=False)

        return DirectionChangeResult(
            cosine_similarity=cos_sim,
            polygon_area=poly_area
        )
