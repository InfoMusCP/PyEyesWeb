Kinetic Energy

Kinetic energy estimation for multi-joint movement.

This module provides a simplified Kinetic Energy feature, computing the weighted sum of squared velocities across multiple joints. It is used to quantify the overall intensity or "vigour" of motion.

KineticEnergy

Bases: StaticFeature

Compute weighted kinetic energy from joint velocities.

Kinetic energy is defined here as:

\[ E_k = \sum_{i=1}^{N} w_i \cdot \|v_i\|^2 \]

where \(w_i\) is the weight (mass) of joint \(i\) and \(v_i\) is its velocity vector.

Read more in the User Guide.

Parameters:
  • weights (list, dict, or None, default: None ) –

    Weights applied to each joint.

    • list or numpy.ndarray — per-joint weights in order.
    • dict — mapping of joint indices to weights.
    • None — equal weights (1.0) for all joints.
  • return_per_joint (bool, default: False ) –

    If True, the result will include the individual energy contribution of each joint. Defaults to False.

Examples:

>>> ke = KineticEnergy(weights=[1.0, 2.0, 1.0], return_per_joint=True)

class KineticEnergy(StaticFeature):
    r"""Compute weighted kinetic energy from joint velocities.

    Kinetic energy is defined here as:

    $$
    E_k = \sum_{i=1}^{N} w_i \cdot \|v_i\|^2
    $$

    where $w_i$ is the weight (mass) of joint $i$ and $v_i$ is its
    velocity vector.

    Read more in the [User Guide](../../user_guide/theoretical_framework/low_level/low_level.md).

    Parameters
    ----------
    weights : list, dict, or None, optional
        Weights applied to each joint.

        * `list` or `numpy.ndarray` — per-joint weights in order.
        * `dict` — mapping of joint indices to weights.
        * `None` — equal weights `(1.0)` for all joints.

    return_per_joint : bool, optional
        If `True`, the result will include the individual energy contribution
        of each joint.  Defaults to `False`.

    Examples
    --------
    >>> ke = KineticEnergy(weights=[1.0, 2.0, 1.0], return_per_joint=True)
    """

    def __init__(self,
                 weights: Union[List[float], Dict[int, float], np.ndarray] = None,
                 return_per_joint: bool = False):
        super().__init__()
        self.weights = weights
        self.return_per_joint = return_per_joint

    def _parse_weights(self, n_signals: int) -> np.ndarray:
        if self.weights is None:
            return np.ones(n_signals)

        if isinstance(self.weights, (list, np.ndarray)):
            w = np.asarray(self.weights)
            if len(w) != n_signals:
                return np.ones(n_signals)  # Fallback to unit weights
            return w

        if isinstance(self.weights, dict):
            w = np.ones(n_signals)
            for idx, val in self.weights.items():
                if 0 <= idx < n_signals:
                    w[idx] = val
            return w

        return np.ones(n_signals)

    def compute(self, frame_data: np.ndarray) -> KineticEnergyResult:
        """Compute kinetic energy for a single frame of velocities.

        Parameters
        ----------
        frame_data : numpy.ndarray of shape (N_signals, N_dims)
            Velocity vectors for each joint in the current snapshot.

        Returns
        -------
        KineticEnergyResult
            Result containing total and (optionally) per-joint energy.
        """
        n_signals = frame_data.shape[0]
        w = self._parse_weights(n_signals)

        # Squared velocity magnitude per joint: ||v||^2
        v_sq = np.sum(frame_data**2, axis=1)

        energies = w * v_sq
        total_energy = np.sum(energies)

        return KineticEnergyResult(
            kinetic_energy=float(total_energy),
            per_joint_energy=energies.tolist() if self.return_per_joint else None
        )

@dataclass(slots=True)
class KineticEnergyResult(FeatureResult):
    """Output contract for the Kinetic Energy feature.

    Attributes
    ----------
    kinetic_energy : float
        Total weighted kinetic energy of the joint configuration.
    per_joint_energy : list of float or None
        Individual energy contributions from each joint.  Populated only if
        `return_per_joint` was enabled in [KineticEnergy][pyeyesweb.low_level.kinetic_energy.KineticEnergy].
    """

    kinetic_energy: float = 0.0
    per_joint_energy: Optional[List[float]] = None

    def to_flat_dict(self, prefix: str = "") -> Dict[str, float]:
        """Flatten joint-wise energies into numbered scalar keys.

        Each element in `per_joint_energy` is exploded into
        `energy_0`, `energy_1`, … entries for flat serialization.

        Parameters
        ----------
        prefix : str, optional
            String prepended to every key.  Defaults to `""`.

        Returns
        -------
        dict of str to float
            Flat mapping of energy keys to scalar values.
        """
        flat = FeatureResult.to_flat_dict(self, prefix)

        def make_key(base: str, idx: int) -> str:
            return f"{prefix}_{base}_{idx}" if prefix else f"{base}_{idx}"

        vals = flat.pop(f"{prefix}_per_joint_energy" if prefix else "per_joint_energy", None)
        if vals is not None:
            for i, val in enumerate(vals):
                flat[make_key("energy", i)] = float(val)

        return flat
