Statistical Moment

Statistical moment extraction from motion signal windows.

This module provides real-time computation of standard statistical moments — mean, standard deviation, skewness, and kurtosis — from windowed motion data. Each metric is computed independently per signal dimension (axis 0 is time), producing a vector of values across features.

Typical use cases include:

  • Characterising the amplitude distribution of joint velocities.
  • Building feature vectors for machine learning pipelines.
  • Detecting asymmetric or heavy-tailed movement distributions.

class StatisticalMoment(DynamicFeature):
    """Real-time statistical moments analyser for windowed signal data.

    Computes one or more statistical moments (mean, standard deviation,
    skewness, kurtosis) over a sliding window of motion data.  Each metric is
    computed **per feature dimension** (i.e. per signal×dim column).

    Read more in the [User Guide](../../user_guide/theoretical_framework/analysis_primitives/analysis_primitives.md).

    Parameters
    ----------
    metrics : list of {"mean", "std_dev", "skewness", "kurtosis"}, optional
        Ordered list of metrics to compute.  Defaults to `["mean"]` when
        `None` is passed.

    Raises
    ------
    ValueError
        If any string in `metrics` is not one of the allowed values.

    Examples
    --------
    >>> sm = StatisticalMoment(metrics=["mean", "std_dev"])
    >>> sm = StatisticalMoment()          # default: mean only
    """

    _ALLOWED_METRICS = ["mean", "std_dev", "skewness", "kurtosis"]
    _MIN_SAMPLES = {"mean": 1, "std_dev": 2, "skewness": 3, "kurtosis": 4}

    def __init__(self, metrics: List[Literal["mean", "std_dev", "skewness", "kurtosis"]] = None):
        super().__init__()
        self.metrics = metrics

    @property
    def metrics(self) -> List[str]:
        """List of moment names to compute."""
        return self._metrics

    @metrics.setter
    def metrics(self, value: Optional[List[str]]):
        target_metrics = value or ["mean"]
        self._metrics = [validate_string(m, self._ALLOWED_METRICS) for m in target_metrics]

    def compute(self, window_data: np.ndarray, **kwargs) -> StatisticalMomentResult:
        """Compute requested statistical moments over the window.

        Parameters
        ----------
        window_data : numpy.ndarray of shape (Time, N_signals, N_dims)
            Window of motion data.  The array is reshaped to
            `(Time, N_signals * N_dims)` and moments are computed along the
            time axis (axis=0).
        **kwargs
            Unused; accepted for API compatibility.

        Returns
        -------
        StatisticalMomentResult
            Result whose list fields correspond to the requested metrics.  A
            field is populated only when enough samples are available to compute
            it (e.g. `std_dev` requires at least 2 samples).  Returns
            `StatisticalMomentResult(is_valid=False)` if no metric could be
            computed.

        Notes
        -----
        Minimum sample requirements per metric:

        * `mean` — 1 sample
        * `std_dev` — 2 samples (ddof=1)
        * `skewness` — 3 samples
        * `kurtosis` — 4 samples
        """
        # Reshape to (Time, Features)
        data = window_data.reshape(window_data.shape[0], -1)
        n_samples = data.shape[0]

        results = {}
        for metric in self._metrics:
            if n_samples >= self._MIN_SAMPLES[metric]:
                if metric == "mean":
                    results["mean"] = np.mean(data, axis=0).tolist()
                elif metric == "std_dev":
                    # ddof=1 for sample standard deviation
                    results["std_dev"] = np.std(data, axis=0, ddof=1).tolist()
                elif metric == "skewness":
                    results["skewness"] = stats.skew(data, axis=0).tolist()
                elif metric == "kurtosis":
                    results["kurtosis"] = stats.kurtosis(data, axis=0).tolist()

        if not results:
            return StatisticalMomentResult(is_valid=False)

        return StatisticalMomentResult(is_valid=True, **results)

@dataclass(slots=True)
class StatisticalMomentResult(FeatureResult):
    """Output contract for the Statistical Moments feature.

    All fields are optional lists; a field is `None` when the corresponding
    metric was not requested or when insufficient samples were available.

    Attributes
    ----------
    mean : list of float or None
        Per-feature sample mean.
    std_dev : list of float or None
        Per-feature sample standard deviation (ddof=1).
    skewness : list of float or None
        Per-feature Fisher–Pearson skewness.
    kurtosis : list of float or None
        Per-feature excess kurtosis (Fisher's definition, normal → 0).
    """

    mean: Optional[List[float]] = None
    std_dev: Optional[List[float]] = None
    skewness: Optional[List[float]] = None
    kurtosis: Optional[List[float]] = None

    def to_flat_dict(self, prefix: str = "") -> Dict[str, float]:
        """Flatten list metrics into numbered scalar keys.

        Each list metric is exploded into `<metric>_0`, `<metric>_1`, …
        entries so that the result can be merged into a flat feature vector.

        Parameters
        ----------
        prefix : str, optional
            String prepended to every key, separated by `_`.  Defaults to
            `""` (no prefix).

        Returns
        -------
        dict of str to float
            Flat mapping of metric-index keys to scalar values.

        Examples
        --------
        >>> result = StatisticalMomentResult(mean=[1.0, 2.0])
        >>> result.to_flat_dict()
        {'is_valid': True, 'mean_0': 1.0, 'mean_1': 2.0}
        """
        # FeatureResult.to_flat_dict bypasses the super() slots bug!
        flat = FeatureResult.to_flat_dict(self, prefix)

        def make_key(base: str, idx: int) -> str:
            return f"{prefix}_{base}_{idx}" if prefix else f"{base}_{idx}"

        # Unroll list properties dynamically
        for metric in ["mean", "std_dev", "skewness", "kurtosis"]:
            key = f"{prefix}_{metric}" if prefix else metric
            vals = flat.pop(key, None)
            if vals is not None:
                for i, val in enumerate(vals):
                    flat[make_key(metric, i)] = float(val)

        return flat
