Results

Feature result data models for PyEyesWeb.

This module provides the standard structure for all feature outputs. All features return a FeatureResult object, which includes validity flags and serialization methods for flat dictionary formats.

@dataclass(slots=True)
class FeatureResult:
    """Standard output contract for all PyEyesWeb features.

    Every feature computation returns a subclass of `FeatureResult`.  The
    base class provides basic validity tracking and flat serialization.

    Read more in the [User Guide](../../user_guide/theoretical_framework/theoretical_framework.md).

    Attributes
    ----------
    is_valid : bool
        ``True`` if the feature was computed successfully.  ``False`` if
        computation failed or insufficient data was available in the window.
    """

    is_valid: bool = True

    def to_flat_dict(self, prefix: str = "") -> Dict[str, Any]:
        """Convert the result to a flat dictionary for logging or CSV export.

        Flattens the dataclass fields into a single-level mapping.  Subclasses
        may override this to provide specialized flattening for complex fields
        (e.g. lists).

        Parameters
        ----------
        prefix : str, optional
            String prepended to every key in the dictionary, separated by
            a `_`.  Defaults to `""`.

        Returns
        -------
        dict
            A flat dictionary with string keys and scalar or list values.
        """
        data = asdict(self)
        if not prefix:
            return data

        return {f"{prefix}_{k}": v for k, v in data.items()}

def to_flat_dict(results: Dict[str, FeatureResult]) -> Dict[str, Any]:
    """Merge multiple feature results into a single flat dictionary.

    Convenience utility for aggregating results from many features into
    one data structure (e.g. for a single row in a results table).

    Parameters
    ----------
    results : dict of str to FeatureResult
        Mapping from feature names (prefixes) to their result objects.

    Returns
    -------
    dict
        A single flat dictionary where each key is prefixed by its
        feature name.

    Examples
    --------
    >>> results = {'smoothness': SmoothnessResult(sparc=-3.5)}
    >>> to_flat_dict(results)
    {'smoothness_is_valid': True, 'smoothness_sparc': -3.5}
    """
    flat = {}
    for name, result in results.items():
        flat.update(result.to_flat_dict(prefix=name))
    return flat
